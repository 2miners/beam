file(GLOB SOURCES "*.cpp")
file(GLOB PARSE_ENCRYPTED_SOURCES "parse_encrypted_keys/*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/wallet/*.hpp")

add_library(libbitcoin_wallet OBJECT ${SOURCES} ${PARSE_ENCRYPTED_SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4244")
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4245")
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4505")
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4458")
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4996")
    target_compile_options(libbitcoin_wallet PUBLIC "/wd4251")
elseif(APPLE)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-mismatched-tags)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-unused-variable)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-missing-braces)
elseif(ANDROID)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-unused-const-variable)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-tautological-constant-compare)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-unused-variable)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-mismatched-tags)
else()
    target_compile_options(libbitcoin_wallet PUBLIC -fpermissive)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_wallet PUBLIC -Wno-unused-but-set-variable)
