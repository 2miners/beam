file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/config/*.hpp")

add_library(libbitcoin_config OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_config PUBLIC "/wd4996")
    target_compile_options(libbitcoin_config PUBLIC "/wd4251")
    target_compile_options(libbitcoin_config PUBLIC "/wd4273")
    #target_compile_options(libbitcoin_config PRIVATE "/wd2491")
elseif(APPLE)
    target_compile_options(libbitcoin_config PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_config PUBLIC -Wno-delete-non-virtual-dtor)
elseif(ANDROID)
    target_compile_options(libbitcoin_config PUBLIC -Wno-delete-non-virtual-dtor)
    target_compile_options(libbitcoin_config PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_config PUBLIC -Wno-tautological-constant-out-of-range-compare)
else()
    target_compile_options(libbitcoin_config PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_config PUBLIC -fpermissive)
endif()
