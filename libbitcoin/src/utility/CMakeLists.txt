file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/utility/*.hpp")

add_library(libbitcoin_utility OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_utility PUBLIC "/wd4244")
    target_compile_options(libbitcoin_utility PUBLIC "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_utility PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_utility PUBLIC -Wno-unused-variable)
elseif(ANDROID)
    target_compile_options(libbitcoin_utility PUBLIC -Wno-tautological-constant-compare)
    target_compile_options(libbitcoin_utility PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_utility PUBLIC -Wno-unused-variable)
else()
    target_compile_options(libbitcoin_utility PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_message PUBLIC -fpermissive)
endif()
