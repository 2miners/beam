file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/utility/*.hpp")

add_library(libbitcoin_utility OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_utility PRIVATE "/wd4244")
    target_compile_options(libbitcoin_utility PRIVATE "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_utility PRIVATE -Wno-reorder)
    target_compile_options(libbitcoin_utility PRIVATE -Wno-unused-variable)
else()
    target_compile_options(libbitcoin_utility PRIVATE -Wno-reorder)
    target_compile_options(libbitcoin_message PRIVATE -fpermissive)
endif()
