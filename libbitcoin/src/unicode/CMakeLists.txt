file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/unicode/*.hpp")

add_library(libbitcoin_unicode OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_unicode PUBLIC "/wd4389")
    target_compile_options(libbitcoin_unicode PUBLIC "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-unused-private-field)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-unused-variable)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-reorder)
elseif(ANDROID)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-unused-private-field)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-unused-variable)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-tautological-constant-out-of-range-compare)
else()
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-reorder)
    target_compile_options(libbitcoin_unicode PUBLIC -Wno-sign-compare)
    target_compile_options(libbitcoin_unicode PUBLIC -fpermissive)
endif()
