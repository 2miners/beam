file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/unicode/*.hpp")

add_library(libbitcoin_unicode OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_unicode PRIVATE "/wd4389")
    target_compile_options(libbitcoin_unicode PRIVATE "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_unicode PRIVATE -Wno-unused-private-field)
    target_compile_options(libbitcoin_unicode PRIVATE -Wno-unused-variable)
    target_compile_options(libbitcoin_unicode PRIVATE -Wno-reorder)
else()
    target_compile_options(libbitcoin_unicode PRIVATE -Wno-reorder)
    target_compile_options(libbitcoin_unicode PRIVATE -Wno-sign-compare)
    target_compile_options(libbitcoin_message PRIVATE -fpermissive)
endif()
