file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/message/*.hpp")

add_library(libbitcoin_message OBJECT ${SOURCES} ${HEADERS})

if(MSVC)
    target_compile_options(libbitcoin_message PUBLIC "/wd4245")
    target_compile_options(libbitcoin_message PUBLIC "/wd4505")
    target_compile_options(libbitcoin_message PUBLIC "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_message PUBLIC -Wno-mismatched-tags)
    target_compile_options(libbitcoin_message PUBLIC -Wno-missing-braces)
elseif(ANDROID)
    target_compile_options(libbitcoin_message PUBLIC -Wno-tautological-constant-compare)
    target_compile_options(libbitcoin_message PUBLIC -Wno-mismatched-tags)
else()
    target_compile_options(libbitcoin_message PUBLIC -fpermissive)
endif()
