file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "${LIBBITCOIN_INCLUDE_DIR}/machine/*.hpp")
add_library(libbitcoin_machine OBJECT ${SOURCES} ${MACHINE})

if(MSVC)
    target_compile_options(libbitcoin_machine PUBLIC "/wd4245")
    target_compile_options(libbitcoin_machine PUBLIC "/wd4505")
    target_compile_options(libbitcoin_machine PUBLIC "/wd4996")
elseif(APPLE)
    target_compile_options(libbitcoin_machine PUBLIC -Wno-mismatched-tags)
    target_compile_options(libbitcoin_machine PUBLIC -Wno-missing-braces)
elseif(ANDROID)
    target_compile_options(libbitcoin_machine PUBLIC -Wno-tautological-constant-compare)
    target_compile_options(libbitcoin_machine PUBLIC -Wno-mismatched-tags)
else()
    target_compile_options(libbitcoin_machine PUBLIC -fpermissive)
endif()
